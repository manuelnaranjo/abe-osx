Based on several instructions on the web, and ABE tools from Linaro I've been able to rebuild gcc-linaro-5.2-2015.11-2 for OSX on ElCapitan. Shortly a tar.xz binary package will be available for download. In the meantime the instructions I used are:

brew install bash
brew install autogen
brew install dejagnu
brew install coreutils
brew install gcc
brew install gawk

git clone https://github.com/manuelnaranjo/abe-osx.git abe

# from crostoolWorks
DMGSIZE="20g"
DMGOUT="disk.dmg"
MPOINT="build"
PKNAME="linaro"
mkdir -p ${MPOINT} 
hdiutil create ${DMGOUT} -ov -volname "${PKNAME}" -fs "Case-sensitive Journaled HFS+" -type UDIF -size "${DMGSIZE}"
hdiutil attach -mountpoint ${MPOINT} ${DMGOUT}

cd ${MPOINT}

# temporarily mess with my system
/usr/local/bin/bash ../abe/osx-prepare.sh

# configure abe
/usr/local/bin/bash ../abe/configure --with-fileserver=148.251.136.42 --with-remote-snapshots=/snapshots-ref

# edit hosts.conf to reduce the number of cpus otherwise you kill your system

# build with the same values of gcc-linaro-5.2-2015.11-2 manifest
/usr/local/bin/bash ../abe/abe.sh --target arm-linux-gnueabihf --build all \
   --set libc=glibc \
   gcc=gcc-linaro-5.2-2015.11-2.tar.xz \
   binutils=binutils-gdb.git~linaro_binutils-2_25-branch@ef90a4718f535cbe6345b4e7168baea7b1972abf \
   gdb=binutils-gdb.git~gdb-7.10-branch@cec9b4821711367dbefd51386eb663ba30905aa6 \
   glibc=glibc.git~release/2.21/master@dbcaca73cb0a19698ea1b424087e8997a9b7c3c4 \
   --tarbin --parallel

# revert my system
/usr/local/bin/bash ../abe/osx-cleanup.sh
